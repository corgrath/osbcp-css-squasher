018    package com.osbcp.csssquasher;
019    
020    import java.util.List;
021    
022    import com.osbcp.cssparser.Rule;
023    
024    /**
025     * Result object from squashing.
026     * 
027     * @author <a href="mailto:christoffer@christoffer.me">Christoffer Pettersson</a>
028     */
029    
030    public class ResultObject {
031    
032            private String originalCSS;
033            private String refactoredCSS;
034            private String compressedCSS;
035            private String log;
036            private List<Rule> rules;
037    
038            /**
039             * Create a new result object.
040             * 
041             * @param originalCSS The original CSS from the input source.
042             * @param refactoredCSS The refactored CSS done by the squasher.
043             * @param compressedCSS The compressed CSS done by YUI Compressor.
044             * @param log The log of the squash process.
045             * @param rules The CSS rules parsed from the input String.
046             */
047    
048            ResultObject(final String originalCSS, final String refactoredCSS, final String compressedCSS, final String log, final List<Rule> rules) {
049                    this.originalCSS = originalCSS;
050                    this.refactoredCSS = refactoredCSS;
051                    this.compressedCSS = compressedCSS;
052                    this.log = log;
053                    this.rules = rules;
054            }
055    
056            /**
057             * Returns the original CSS from the input source.
058             * 
059             * @return The original CSS from the input source.
060             */
061    
062            public String getOriginalCSS() {
063                    return originalCSS;
064            }
065    
066            /**
067             * Returns the refactored CSS done by the squasher.
068             * 
069             * @return The refactored CSS done by the squasher.
070             */
071    
072            public String getRefactoredCSS() {
073                    return refactoredCSS;
074            }
075    
076            /**
077             * Returns the compressed CSS done by YUI Compressor.
078             * 
079             * @return The compressed CSS done by YUI Compressor.
080             */
081    
082            public String getCompressedCSS() {
083                    return compressedCSS;
084            }
085    
086            /**
087             * Returns the log of the squash process.
088             * 
089             * @return the log of the squash process.
090             */
091    
092            public String getLog() {
093                    return log;
094            }
095    
096            /**
097             * Returns the CSS rules parsed from the input String. 
098             * 
099             * @return The CSS rules parsed from the input String.
100             */
101    
102            public List<Rule> getRules() {
103                    return rules;
104            }
105    
106    }
